Compliant Solution (AsyncAPI 2)

    asyncapi: '2.4.0'
    info:
      title: Unique Operation ID Example API
      version: '1.0.0'
      description: API que demuestra el uso de operationId único en cada operación.
    channels:
      /pets:
        description: Canal para operaciones relacionadas con mascotas.
        subscribe: 
          operationId: getPets
          summary: summary example
          message:
            contentType: application/json
            payload:
              type: array
              items:
                type: object
                properties:
                  id:
                    type: string
                    description: ID de la mascota.
                  name:
                    type: string
                    description: Nombre de la mascota.

                    type: string

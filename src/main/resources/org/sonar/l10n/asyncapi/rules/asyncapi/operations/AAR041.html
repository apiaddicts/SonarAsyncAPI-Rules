Compliant Solution (AsyncAPI 2)

asyncapi: '2.4.0'
info:
  title: Compliant API
  version: '1.0.0'
  description: This API follows the recommendation to add servers and channels to components.
  license:
    name: Apache 2.0
    url: https://www.apache.org/licenses/LICENSE-2.0.html

components:
  servers:
    production:
      url: api.example.com
      protocol: https
      description: Production server
  channels:
    pets:
      description: Channel for managing pet data
      subscribe:
        summary: Subscribe to pet updates
        operationId: receivePetUpdates
        message:
          contentType: application/json
          payload:
            type: object
            properties:
              id:
                type: string
                description: ID of the pet
              name:
                type: string
                description: Name of the pet
